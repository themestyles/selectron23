function add_currency(value, num = 1) {

    let selector = null;
    if (parseInt(num) === 1) {
        selector = selector1;
    } else {
        selector = selector2;
    }

    if (selector.append({
        value: value,
        title: 'JPY',
        text: 'Japan Yen',
        img: 'https://pluginus.net/wp-content/uploads/2021/03/japan.gif'
    })) {
        selector.select(value);
    } else {
        alert('JPY already in!');
}
}
